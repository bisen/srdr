def self.has_study_data(id)
	datapoints = DesignDetailDataPoint.where(:design_detail_field_id => id).all
	if datapoints.nil? || (datapoints.length == 0)
		return "false"
	else
		return "true"
	end
end

def get_subquestion
	fields = self.design_detail_fields
	subq = nil
	unless fields.empty?
		fields.each do |field|
		  if field.has_subquestion == true
		  	subq = field.subquestion
		  	break;
		  end
		end
	end
	return subq
end

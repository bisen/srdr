Overview

This controller handles creation, deletion, pubmed data, and updating of primary publications for a study. A study has one primary publication, and a primary publication can have many primary publication numbers. The study title is derived from the title saved in the primary publication.

def update
  @primary_publication = PrimaryPublication.find(params[:id])
@study = Study.find(@primary_publication.study_id)	

	# update any changes to previously specified publication numbers
	@ppi_ids = []
	if !params[:primary_publication_numbers_attributes].nil?
	params[:primary_publication_numbers_attributes].each do |item|
@existing_id = item[1][:id]
@ppi_ids << @existing_id
@ppi = PrimaryPublicationNumber.find(@existing_id)
@ppi.number = item[1][:number]
@ppi.number_type = item[1][:number_type]
@ppi.primary_publication_id = @primary_publication.id
@ppi.save
	end
	end
	
	@ppis_to_delete = PrimaryPublicationNumber.where(["primary_publication_id=(?) AND id NOT IN (?)", @primary_publication.id, @ppi_ids]).all
	@ppis_to_delete.each do |tp|
tp.destroy
	end
	
	# create newly specified outcomes
	if !params[:primary_publication_numbers_attributes].nil?
	params[:primary_publication_numbers_attributes].each do |item|
@ppn = PrimaryPublicationNumber.new
@ppn.number = item[1][:number]
@ppn.number_type = item[1][:number_type]
@ppn.primary_publication_id = @primary_publication.id
@ppn.save	
	end			
	end
	

  if @saved = @primary_publication.update_attributes(params[:primary_publication])
	@message_div = "saved_indicator_1"
else
   	problem_html = create_error_message_html(@primary_publication.errors)
	flash[:modal_error] = problem_html
	@error_partial = "layouts/info_messages"	
	@error_div = "validation_message"	
  end

end

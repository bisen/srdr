Overview

This controller handles creation, deletion, and updating of users. The user class is linked to the authlogic authentication gem. This enables authentication/login and logout, password encryption and password resetting. For more information see: github.com/binarylogic/authlogic For specifics about setting up authlogic on rails 3, see www.dixis.com/?p=352

def create
   @user = User.new(params[:user])
   @user.user_type = params[:user][:user_type]
   @user.email.downcase!
   @user.login.downcase!
   if @user.save
     flash[:success_message] = "Registration was successful."
     redirect_back_or_default "/"
   else
   	msg = "<strong>Registration Unsuccessful:</strong> "
   	@user.errors.keys.each do |key|
   		@user.errors[key].each do |reason|
   			msg += "<br/><strong>#{key}</strong>: #{reason}"
   		end
   	end
     flash[:error_message] = "#{msg}"
     redirect_to :back
   end
end

def update
  @user = @current_user # makes our views "cleaner" and more consistent
  params[:user][:email].downcase!
  if @user.update_attributes(params[:user])
    flash[:notice] = "Account updated!"
    redirect_to account_url
  else
    render :action => :edit
  end
end
